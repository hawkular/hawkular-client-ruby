def hawk_escape(url_part)
  return url_part.to_s if url_part.is_a?(Numeric)

  if url_part.is_a? Symbol
    sub_url = url_part.to_s
  else
    sub_url = url_part.dup
  end
  sub_url.gsub!('%', '%25')
  sub_url.gsub!(' ', '%20')
  sub_url.gsub!('[', '%5b')
  sub_url.gsub!(']', '%5d')
  sub_url.gsub!('|', '%7c')
  sub_url.gsub!('(', '%28')
  sub_url.gsub!(')', '%29')
  sub_url.gsub!('/', '%2f')
  sub_url
end

def hawk_escape_id(url_part)
  sub_url = hawk_escape url_part
  sub_url.gsub!('=', '%3d')
  sub_url.gsub!(';', '%3b')
  sub_url
end
