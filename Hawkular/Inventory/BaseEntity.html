def initialize(hash)
  @id = hash['id']
  @path = hash['path']
  @name = hash['name'] || @id
  @properties = hash['properties'] || {}
  @_hash = hash.dup

  return if @path.nil?

  tmp = @path.split('/')
  tmp.each do |pair|
    (key, val) = pair.split(';')
    case key
    when 'f'
      @feed = val
    when 'e'
      @env = val
    end
  end
end
